### main.rs

```rust
#[macro_use] extern crate nickel;

use nickel::Nickel;

fn main() {
    let mut server = Nickel::new();

    server.utilize(router! {
        get "**" => |_req, _res| { "Hello world!"}
    });

    // only localhost
    // server.listen("127.0.0.1:6767").unwrap();
    
    // visible on LAN 
    server.listen("0.0.0.0:6767").unwrap();
}
```
