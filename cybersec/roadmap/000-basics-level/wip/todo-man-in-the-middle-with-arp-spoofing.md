### scan all your LAN

```bash
#! /bin/bash

echo "change your MAC =D"

ip=$(hostname -I | awk '{print $1}')
ip="${ip##*( )}"
ip_selected=()

if [[ "$ip" == *"$192"* ]]; then
    ip_selected=192.168.0.0/16
elif [[ "$ip" == *"$172"* ]]; then
    ip_selected=172.16.0.0/12
elif [[ "$ip" == *"$10"* ]]; then
    ip_selected=10.0.0.0/8
fi

# echo "$ip_selected"
nmap -sL "$ip_selected" 2>&1 | tee | grep --line-buffered -e \( | awk -Ffor '{print $2}'
```
